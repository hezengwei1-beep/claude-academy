## 嵌入 Next.js 应用

### API Route

```typescript
// app/api/ai/review/route.ts
import { createAgent } from '@anthropic-ai/claude-code';
import { NextResponse } from 'next/server';

export async function POST(req: Request) {
  const { code, language } = await req.json();

  const agent = createAgent({
    model: 'claude-sonnet-4-6',
    systemPrompt: '你是代码审查专家。输出 JSON 格式。',
    maxTokens: 50000,
  });

  const result = await agent.run(`
    审查以下 ${language} 代码，输出 JSON：
    {"issues": [{"severity": "high|medium|low", "line": N, "description": "..."}]}

    代码：
    ${code}
  `);

  return NextResponse.json({ review: result.output });
}
```

### 前端调用

```typescript
// components/CodeReviewer.tsx
async function reviewCode(code: string) {
  const res = await fetch('/api/ai/review', {
    method: 'POST',
    body: JSON.stringify({ code, language: 'typescript' }),
  });
  const data = await res.json();
  return JSON.parse(data.review);
}
```

## 构建 CLI 工具

```typescript
#!/usr/bin/env node
// bin/my-review-tool.ts

import { createAgent } from '@anthropic-ai/claude-code';
import { execSync } from 'child_process';

const diff = execSync('git diff --staged').toString();

if (!diff) {
  console.log('没有暂存的改动');
  process.exit(0);
}

const agent = createAgent({
  model: 'claude-sonnet-4-6',
  systemPrompt: '代码审查。只报告问题，不废话。',
  maxTokens: 50000,
});

const result = await agent.run(`审查这些改动：\n${diff}`);
console.log(result.output);

// 退出码：有 Critical 问题返回 1
if (result.output.includes('Critical')) {
  process.exit(1);
}
```

## 构建 Slack Bot

```typescript
import { App } from '@slack/bolt';
import { createAgent } from '@anthropic-ai/claude-code';

const slackApp = new App({
  token: process.env.SLACK_BOT_TOKEN,
  signingSecret: process.env.SLACK_SIGNING_SECRET,
});

slackApp.message('review', async ({ message, say }) => {
  const agent = createAgent({
    model: 'claude-sonnet-4-6',
    tools: ['Read', 'Grep'],
    workingDirectory: '/path/to/project',
  });

  const result = await agent.run(`审查最近的 git diff，生成简要报告`);
  await say(result.output);
});

slackApp.start(3000);
```

## CI/CD Pipeline

```typescript
// scripts/ci-review.ts
import { createAgent, createSubAgent } from '@anthropic-ai/claude-code';

async function ciReview() {
  const diff = execSync('git diff origin/main...HEAD').toString();

  // 并行审查
  const [security, perf, logic] = await Promise.all([
    createSubAgent({ name: 'security', model: 'claude-sonnet-4-6' })
      .run(`安全审查：\n${diff}`),
    createSubAgent({ name: 'perf', model: 'claude-sonnet-4-6' })
      .run(`性能审查：\n${diff}`),
    createSubAgent({ name: 'logic', model: 'claude-sonnet-4-6' })
      .run(`逻辑审查：\n${diff}`),
  ]);

  // 汇总
  const lead = createAgent({ model: 'claude-sonnet-4-6' });
  const report = await lead.run(`
    综合审查结果，输出 JSON：
    安全：${security.output}
    性能：${perf.output}
    逻辑：${logic.output}
  `);

  const issues = JSON.parse(report.output);
  const critical = issues.filter(i => i.severity === 'critical');

  if (critical.length > 0) {
    console.error('发现 Critical 问题:', critical);
    process.exit(1);  // CI 失败
  }

  console.log('审查通过');
}

ciReview();
```
