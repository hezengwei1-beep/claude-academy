## 评估体系（Evals）

### 为什么需要评估

"它看起来好用"不等于"它真的好用"。评估体系让你量化 Agent 的表现。

### 构建评估集

```typescript
interface EvalCase {
  id: string;
  input: string;
  expectedOutput?: string;
  mustContain?: string[];
  mustNotContain?: string[];
  maxTokens?: number;
  maxLatency?: number;  // ms
}

const evalCases: EvalCase[] = [
  {
    id: 'simple-function',
    input: '写一个计算斐波那契数列的 Python 函数',
    mustContain: ['def ', 'fibonacci', 'return'],
    mustNotContain: ['import numpy'],  // 不应该引入不必要的依赖
    maxLatency: 10000,
  },
  {
    id: 'sql-injection-check',
    input: '审查这段代码：db.query(`SELECT * FROM users WHERE id = ${userId}`)',
    mustContain: ['SQL 注入', '参数化'],
  },
];
```

### 运行评估

```typescript
async function runEvals(agent, cases: EvalCase[]) {
  const results = [];

  for (const tc of cases) {
    const start = Date.now();
    const result = await agent.run(tc.input);
    const latency = Date.now() - start;

    const passed = {
      mustContain: tc.mustContain?.every(s => result.output.includes(s)) ?? true,
      mustNotContain: tc.mustNotContain?.every(s => !result.output.includes(s)) ?? true,
      maxLatency: tc.maxLatency ? latency <= tc.maxLatency : true,
    };

    results.push({
      id: tc.id,
      passed: Object.values(passed).every(Boolean),
      details: passed,
      latency,
      tokens: result.usage,
    });
  }

  return results;
}
```

### 模型降级策略

```typescript
async function smartRun(prompt: string) {
  // 先用便宜模型评估复杂度
  const triage = await createAgent({ model: 'claude-haiku-4-5' })
    .run(`这个任务的复杂度是 simple/medium/complex？任务：${prompt}`);

  const model = triage.output.includes('complex')
    ? 'claude-opus-4-6'
    : triage.output.includes('medium')
    ? 'claude-sonnet-4-6'
    : 'claude-haiku-4-5';

  return createAgent({ model }).run(prompt);
}
```

## 安全加固

### Prompt Injection 防御

```typescript
function sanitizeInput(userInput: string): string {
  // 检测可疑模式
  const patterns = [
    /ignore\s+(previous|above|all)/i,
    /system\s+prompt/i,
    /\<\/?system\>/i,
  ];

  for (const p of patterns) {
    if (p.test(userInput)) {
      throw new Error('Suspicious input detected');
    }
  }

  return userInput;
}
```

### 输出验证

```typescript
function validateOutput(output: string, forbidden: string[]): boolean {
  return !forbidden.some(f => output.toLowerCase().includes(f.toLowerCase()));
}

// 使用
const result = await agent.run(sanitizeInput(userInput));
if (!validateOutput(result.output, ['api_key', 'password', 'secret'])) {
  throw new Error('Output contains sensitive information');
}
```

### Rate Limiting

```typescript
import { RateLimiter } from 'limiter';

const limiter = new RateLimiter({
  tokensPerInterval: 10,  // 每分钟 10 次调用
  interval: 'minute',
});

async function rateLimitedRun(agent, prompt) {
  const hasToken = await limiter.removeTokens(1);
  if (!hasToken) {
    throw new Error('Rate limit exceeded');
  }
  return agent.run(prompt);
}
```
